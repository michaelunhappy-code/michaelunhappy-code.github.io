🏆 最短路算法 C++ 模板库 (完整版)

结合常见的最短路算法及其应用场景（面向竞赛及高性能场景优化）。所有模板均基于 1-based 索引，处理了 INF 溢出问题，适用于各大竞赛平台。

1. BFS (广度优先搜索)

核心思路：基于队列的层序遍历。每次将当前距离+1的节点入队，保证第一次访问到节点时即为最短路。
时间复杂度：$O(V+E)$ （常简写为 $O(E)$）
适用场景：图的边权全为 1（或边权为 0/1 的图，可使用双端队列 `std::deque` 优化）。 
#include <iostream>
#include <vector>
#include <queue>

using namespace std;
const int INF = 1e9;

// 边权为 1 的无向图/有向图最短路
vector<int> bfs(int n, int s, const vector<vector<int>>& adj) {
    vector<int> dist(n + 1, INF);
    queue<int> q;
    
    dist[s] = 0;
    q.push(s);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : adj[u]) {
            if (dist[v] == INF) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
    return dist;
}

2. Dijkstra 算法 (堆优化)

核心思路：贪心策略。每次从未确定最短路的节点中，选出距离起点最近的节点，标记为确定，并用它更新相邻节点的距离。利用 std::priority_queue 优化查找最小值的过程。
时间复杂度：朴素版 $O(V^2)$，可利用数据结构优化至 $O((V+E) \log E)$
适用场景：单源最短路，图中不能有负权边。 
#include <iostream>
#include <vector>
#include <queue>

using namespace std;
const long long INF = 1e18;

struct Edge {
    int to;
    long long weight;
};

vector<long long> dijkstra(int n, int s, const vector<vector<Edge>>& adj) {
    vector<long long> dist(n + 1, INF);
    vector<bool> vis(n + 1, false);
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;
    
    dist[s] = 0;
    pq.push({0, s});
    
    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();
        if (vis[u]) continue;
        vis[u] = true;
        
        for (const auto& edge : adj[u]) {
            int v = edge.to;
            long long w = edge.weight;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}

3. Bellman-Ford 算法 (含负环判定)

核心思路：暴力对所有的边进行 $V-1$ 次松弛操作。如果在进行第 $V$ 次遍历时依然有边可以被松弛，说明图中存在负环。
时间复杂度：$O(VE)$
适用场景：单源最短路，能用来判负环（支持负权边）。 
#include <iostream>
#include <vector>

using namespace std;
const long long INF = 1e18;

struct Edge {
    int u, v;
    long long weight;
};

// 如果存在负环返回空 vector，否则返回最短路数组
vector<long long> bellman_ford(int n, int s, const vector<Edge>& edges) {
    vector<long long> dist(n + 1, INF);
    dist[s] = 0;
    
    for (int i = 1; i < n; ++i) {
        bool relaxed = false;
        for (const auto& edge : edges) {
            if (dist[edge.u] != INF && dist[edge.u] + edge.weight < dist[edge.v]) {
                dist[edge.v] = dist[edge.u] + edge.weight;
                relaxed = true;
            }
        }
        if (!relaxed) break; // 提前结束优化
    }
    
    // 第 V 次松弛判负环
    for (const auto& edge : edges) {
        if (dist[edge.u] != INF && dist[edge.u] + edge.weight < dist[edge.v]) {
            return {}; // 存在负环
        }
    }
    
    return dist;
}

4. SPFA 算法 (队列优化的 Bellman-Ford)

核心思路：仅当一个点的最短路被更新时，它的邻接点才有可能被更新。将更新过的点放入队列中避免无效松弛。同时可通过记录入队次数判断是否存在负环。
时间复杂度：平均 $O(kE)$，最坏退化为 $O(VE)$（容易被特殊构造的图卡死）。
适用场景：包含负权边的单源最短路，以及快速判断负环。 
#include <iostream>
#include <vector>
#include <queue>

using namespace std;
const long long INF = 1e18;

struct AdjEdge {
    int to;
    long long weight;
};

vector<long long> spfa(int n, int s, const vector<vector<AdjEdge>>& adj) {
    vector<long long> dist(n + 1, INF);
    vector<bool> inq(n + 1, false);
    vector<int> cnt(n + 1, 0); 
    queue<int> q;
    
    dist[s] = 0;
    inq[s] = true;
    q.push(s);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        inq[u] = false;
        
        for (const auto& edge : adj[u]) {
            int v = edge.to;
            long long w = edge.weight;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                cnt[v] = cnt[u] + 1;
                if (cnt[v] >= n) return {}; // 判负环
                
                if (!inq[v]) {
                    q.push(v);
                    inq[v] = true;
                }
            }
        }
    }
    return dist;
}

5. Floyd 算法

核心思路：基于动态规划，设 dp[k][i][j] 为只允许经过前 k 个节点时，i 到 j 的最短路。核心循环结构必须是先枚举中转点 $k$，再枚举起点 $i$ 和终点 $j$。
时间复杂度：$O(V^3)$
适用场景：图中不能有负环，能求出任意两点间最短距离。 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
const long long INF = 1e18;

// 传入邻接矩阵，未连接的边权值为 INF，自环权值为 0
void floyd(int n, vector<vector<long long>>& dist) {
    for (int k = 1; k <= n; ++k) {
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= n; ++j) {
                if (dist[i][k] != INF && dist[k][j] != INF) {
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
    }
}
