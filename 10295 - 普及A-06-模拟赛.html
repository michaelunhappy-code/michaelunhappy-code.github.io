A

矩阵递推

题目大意

给定长度为 $n$ 的序列 $l,t$ 作为矩阵 $F$ 的第一列和第一行（$l_1=t_1$），其余元素满足 $$F_{i,j}=a\cdot F_{i,j-1}+b\cdot F_{i-1,j}$$ 求 $F_{n,n} \bmod 10^9+7$。

思路分析

  • 递推式中 $F_{i,j}$ 只依赖同一行左边一格和上一行同列一格，因此可以逐行递推。
  • 直接开 $n\times n$ 矩阵空间为 $O(n^2)$，当 $n=5000$ 时约 2 亿个 long long，会超出空间限制。
  • 滚动数组优化：用 pre[] 保存上一行、dp[] 计算当前行。初始化 dp = t（第一行），每行开头用 dp[1] = l[i]（第一列），随后从左到右转移： dp[j] = a·dp[j-1] + b·pre[j]。
  • 全程取模，最终答案即 dp[n]。

复杂度分析

时间复杂度
$O(n^2)$
双重循环逐行递推，$n=5000$ 时约 $2.5\times10^7$ 次运算。
空间复杂度
$O(n)$
滚动数组，只保留相邻两行，避免 $O(n^2)$ 空间爆炸。

标准程序

#include <bits/stdc++.h>
using namespace std;
const int N=5005;
const long long MOD=1e9+7;
long long n,a,b;
long long dp[N],pre[N],l[N],t[N];
int main(){
    cin>>n>>a>>b;
    for(int i=1;i<=n;i++){
        cin>>l[i];
    }
    for(int i=1;i<=n;i++){
        cin>>t[i];
    }
    for(int i=1;i<=n;i++){
        dp[i]=t[i];                       // 第一行
    }
    for(int i=2;i<=n;i++){
        for(int j=1;j<=n;j++){
            pre[j]=dp[j];                 // 保存上一行
        }
        dp[1]=l[i];                       // 第一列
        for(int j=2;j<=n;j++){
            dp[j]=(a*dp[j-1]%MOD+b*pre[j]%MOD)%MOD;
        }
    }
    cout<<dp[n]%MOD<<endl;
    return 0;
}
B

环上连边

题目大意

$n$ 个点构成环，点有权值 $a_i$。选中的点向其余 $n-1$ 个点全部连边，之后去重。 边权为两端点权值之积，求所有边权之和。

思路分析

  • 去重后，点对 $(i,j)$ 之间有边 $\iff$ 至少一个被选中，或 $i,j$ 在环上相邻。
  • 用正难则反：至少一个被选中的点对贡献 = 所有点对贡献 − 两个都未选的点对贡献。
  • 经典恒等式： $$\sum_{i<j}a_ia_j=\frac{(\sum a_i)^2-\sum a_i^2}{2}$$ 分别对全体（$s,s_2$）和未选点集（$su,su_2$）套用，相减得 ans1。
  • 环上相邻的边中，两端都被选或恰一端被选的已被 ans1 统计过；只需补上两端都未选的相邻点对，即 ans2——遍历环枚举 $(i, i\bmod n+1)$ 即可。
  • 注意 $a_i$ 平方求和可能很大，全程使用 long long。

复杂度分析

时间复杂度
$O(n)$
两次线性扫描：一次统计前缀和，一次枚举环上相邻对。
空间复杂度
$O(n)$
存储权值数组与选择标记。

标准程序

#include <bits/stdc++.h>
using namespace std;
const int N=100010;
long long a[N];
int c[N];
int main(){
    int n;
    cin>>n;
    long long s=0,s2=0,su=0,su2=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        s+=a[i];
        s2+=a[i]*a[i];
    }
    for(int i=1;i<=n;i++){
        cin>>c[i];
        if(c[i]==0){
            su+=a[i];
            su2+=a[i]*a[i];
        }
    }
    // 至少一端被选的点对之和 = 全部点对 - 两端都未选的点对
    long long ans1=(s*s-s2)/2-(su*su-su2)/2;
    long long ans2=0;
    // 补上环上相邻且两端都未选的边
    for(int i=1;i<=n;i++){
        int j=i+1;
        if(j>n) j=1;
        if(c[i]==0&&c[j]==0){
            ans2+=a[i]*a[j];
        }
    }
    cout<<ans1+ans2<<endl;
    return 0;
}
C

位运算询问

题目大意

长度为 $n$ 的数列，$q$ 次操作：① 单点修改 $a_x\leftarrow y$；② 询问 $\sum_{i=1}^n a_i \,\text{and}\, x$。

思路分析

  • 按位与运算各二进制位互相独立，可以逐位拆开统计贡献。
  • 维护 cnt[b] = 数列中第 $b$ 位为 1 的元素个数。则询问答案为： $$\sum_{b:\,x\text{ 第 }b\text{ 位为 }1} 2^b\cdot cnt[b]$$ 因为只有 $x$ 该位为 1 时，与运算才可能保留该位，保留的个数恰为 cnt[b]。
  • 修改：把 $a_x$ 各位从 cnt 中减掉，再把 $y$ 各位加进去，$O(60)$ 完成。
  • 累加时 (1LL<<b) * cnt[b] 必须用 long long，否则移位溢出。
⚠️ 修改点：原代码询问循环为 b <= 25，与统计范围（60 位）不一致——当 $x$ 的高位（bit 26 ~ 59）为 1 时会漏算。标程中已统一改为 b < 60。

复杂度分析

时间复杂度
$O\big((n+q)\cdot 60\big)$
预处理 $O(60n)$，每次修改/询问均为 $O(60)$。
空间复杂度
$O(n + 60)$
原数组 + 每一位的计数桶。

标准程序

#include <bits/stdc++.h>
using namespace std;
const int N=1e5+5;
typedef long long LL;
int n,q;
LL a[N];
LL cnt[61];
int main(){
    cin>>n>>q;
    for(int i=1;i<=n;++i){
        cin>>a[i];
        for(int b=0;b<60;++b){
            if(a[i]>>b&1) cnt[b]++;
        }
    }
    while(q--){
        int id;
        cin>>id;
        if(id==1){
            LL x,y;
            cin>>x>>y;
            for(int b=0;b<60;++b){
                if(a[x]>>b&1) cnt[b]--;   // 撤销旧值贡献
                if(y>>b&1) cnt[b]++;      // 加入新值贡献
            }
            a[x]=y;
        }
        else{
            LL x;
            cin>>x;
            LL ans=0;
            for(int b=0;b<60;++b){        // 修正: 原为 b<=25
                if(x>>b&1){
                    ans += (1LL<<b) * cnt[b];
                }
            }
            cout<<ans<<"\n";
        }
    }
    return 0;
}
D

物品选择

题目大意

$n$ 个物品，价值 $v_i$、代价 $w_i$。以任意顺序选任意数量，每选一个物品 $i$， 剩余物品价值全部减少 $w_i$。最大化选中物品的实际价值之和。

思路分析

  • 第一步——交换论证确定最优顺序：设最终选了某个集合，考察相邻的两次选择 A（先）、B（后）：A 使 B 损失 $w_A$，交换后 B 使 A 损失 $w_B$。故 $w$ 小的先选更优 ⟹ 最优选取顺序为 $w$ 升序。
  • 第二步——排序后 DP：将物品按 $w$ 从大到小排序依次处理。若第 $i$ 个物品被作为已选集合的第 $j$ 个加入，则它的 $w$ 是当前已选中最小的，实际选取时排最前，会让后面 $j-1$ 个物品各损失 $w_i$，总惩罚恰为 $(j-1)\,w_i$。
  • 状态：dp[j] = 已选 $j$ 个物品的最大总价值。转移（倒序枚举 $j$，类 0/1 背包）： $$dp[j]=\max\big(dp[j],\; dp[j-1]+v_i-(j-1)\,w_i\big)$$
  • 初值 dp[0]=0，其余为 $-\infty$（防止非法状态转移）。答案为 $\max_j dp[j]$，与 0 取 max（可以一个都不选）。

复杂度分析

时间复杂度
O(n²)
每个物品进行一次背包式转移，整体 O(n²)
空间复杂度
O(n)
仅使用 dp 数组存储状态

标准程序

#include <bits/stdc++.h>
using namespace std;
const int N=5005;
struct P{
    long long v,w;
}a[N];
long long dp[N];
int n;
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i].v>>a[i].w;
    }
    sort(a+1,a+n+1,[](P x,P y){
        return x.w>y.w;
    });
    for(int i=0;i<=n;i++){
        dp[i]=LLONG_MIN;
    }
    dp[0]=0;
    for(int i=1;i<=n;i++){
        for(int j=i;j>=1;j--){
            dp[j]=max(dp[j], dp[j-1]+a[i].v-(j-1)*a[i].w);
        }
    }
    long long ans=0;
    for(int i=0;i<=n;i++){
        ans=max(ans,dp[i]);
    }
    cout<<ans<<endl;
    return 0;
}
